175inline Version parse_version_string(const std::string& version_string) {
176 int major = 0, minor = 0, patch = 0;
177 size_t i = 0, end = version_string.size();
178
179 // MAJOR VERSION.
180 if (version_string.empty()) {
181 throw std::runtime_error("version string is empty");
182 }
183 if (version_string[i] == '0') {
184 throw std::runtime_error("invalid version string '" + version_string + "' has leading zeros in its major version");
185 }
186 while (i < end && version_string[i] != '.') {
187 if (!std::isdigit(version_string[i])) {
188 throw std::runtime_error("invalid version string '" + version_string + "' contains non-digit characters");
189 }
190 major *= 10;
191 major += version_string[i] - '0';
192 ++i;
193 }
194
195 // MINOR VERSION.
196 if (i == end) {
197 throw std::runtime_error("version string '" + version_string + "' is missing a minor version");
198 }
199 ++i; // get past the period and check again.
200 if (i == end) {
201 throw std::runtime_error("version string '" + version_string + "' is missing a minor version");
202 }
203
204 if (version_string[i] != '0') {
205 while (i < end && version_string[i] != '.') {
206 if (!std::isdigit(version_string[i])) {

208 }
209 minor *= 10;
210 minor += version_string[i] - '0';
211 ++i;
212 }
213
214 } else {
215 ++i;
216 if (i < end && version_string[i] != '.') {
217 throw std::runtime_error("version string '" + version_string + "' has leading zeros in its minor version");
218 }
219 }

228 }
