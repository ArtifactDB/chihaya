128inline ArrayDetails validate(const H5::Group& handle) {
129 ritsuko::Version version;
130
131 if (handle.attrExists("delayed_version")) {
132 auto ahandle = handle.openAttribute("delayed_version");
133 if (!ritsuko::hdf5::is_utf8_string(ahandle)) {
134 throw std::runtime_error("expected 'delayed_version' to use a datatype that can be represented by a UTF-8 encoded string");
135 }
136 auto vstring = ritsuko::hdf5::load_scalar_string_attribute(ahandle);
137 version = ritsuko::parse_version_string(vstring.c_str(), vstring.size());
138 }
139
140 return validate(handle, version);
141}
