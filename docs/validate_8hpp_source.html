57inline ArrayDetails validate(const H5::Group& handle, const std::string& name, const Version& version) {
58 auto dtype = load_string_attribute(handle, "delayed_type", " for a delayed object");
59 ArrayDetails output;
60
61 if (dtype == "array") {
62 try {
63 auto atype = load_string_attribute(handle, "delayed_array", " for an array");
64
65 // Checking external.
66 if (atype == "dense array") {
67 output = validate_dense_array(handle, name, version);
68 } else if (atype == "sparse matrix") {
69 output = validate_sparse_matrix(handle, name, version);
70 } else if (atype.rfind("custom ", 0) != std::string::npos) {
71 output = validate_custom_array(handle, name, version);
72 } else if (atype.rfind("external hdf5 ", 0) != std::string::npos) {
73 output = validate_external_hdf5(handle, name, version);
74 } else if (atype == "constant array") {
75 output = validate_constant_array(handle, name, version);
76 } else {
77 throw std::runtime_error(std::string("unknown array type '") + atype + "' at '" + name + "'");
78 }
79
80 } catch (std::exception& e) {
81 throw std::runtime_error(std::string("failed to validate delayed array at '") + name + "':\n " + e.what());
82 }
83
84 } else if (dtype == "operation") {
85 try {
86 auto otype = load_string_attribute(handle, "delayed_operation", " for an operation");
87
88 // Checking subset.
89 if (otype == "subset") {
90 output = validate_subset(handle, name, version);
91 } else if (otype == "combine") {
92 output = validate_combine(handle, name, version);
93 } else if (otype == "transpose") {
94 output = validate_transpose(handle, name, version);
95 } else if (otype == "dimnames") {
96 output = validate_dimnames(handle, name, version);
97 } else if (otype == "subset assignment") {
98 output = validate_subset_assignment(handle, name, version);
99 } else if (otype == "unary arithmetic") {
100 output = validate_unary_arithmetic(handle, name, version);
101 } else if (otype == "unary comparison") {
102 output = validate_unary_comparison(handle, name, version);
103 } else if (otype == "unary logic") {
104 output = validate_unary_logic(handle, name, version);
105 } else if (otype == "unary math") {
106 output = validate_unary_math(handle, name, version);
107 } else if (otype == "unary special check") {
108 output = validate_unary_special_check(handle, name, version);
109 } else if (otype == "binary arithmetic") {
110 output = validate_binary_arithmetic(handle, name, version);
111 } else if (otype == "binary comparison") {
112 output = validate_binary_comparison(handle, name, version);
113 } else if (otype == "binary logic") {
114 output = validate_binary_logic(handle, name, version);

116 output = validate_matrix_product(handle, name, version);

118 throw std::runtime_error(std::string("unknown operation type '") + otype + "' at '" + name + "'");

143inline ArrayDetails validate(const H5::Group& handle, const std::string& name) {
144 Version version;
145 if (handle.attrExists("delayed_version")) {
146 auto ahandle = handle.openAttribute("delayed_version");
