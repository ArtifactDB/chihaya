178inline ArrayDetails validate(const H5::Group& handle, State& state) {
179 ritsuko::Version version;
180
181 if (handle.attrExists("delayed_version")) {
182 auto ahandle = handle.openAttribute("delayed_version");
183 if (!ritsuko::hdf5::is_utf8_string(ahandle)) {
184 throw std::runtime_error("expected 'delayed_version' to use a datatype that can be represented by a UTF-8 encoded string");
185 }
186
187 auto vstring = ritsuko::hdf5::load_scalar_string_attribute(ahandle);
188 if (vstring == "1.0.0") {
189 version.major = 1;
190 } else {
191 version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
192 }
193 } else {
194 version.minor = 99;
195 }
196
197 return validate(handle, version, state);
198}
